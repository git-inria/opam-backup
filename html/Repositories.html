Module Repositories


module Repositories: sig .. end
Generic repository pluggin

The following functions are wrapper to the corresponding scripts
val init : Types.repository -> unit
Run opam-$kind-init in $opam/repo/$repo
val update : Types.repository -> unit
Run opam-$kind-update in $opam/repo/$repo
val download : Types.repository -> Types.nv -> unit
Run opam-$kind-download in $opam/repo/$repo
val upload : Types.repository -> unit
Run opam-$kind-upload in $opam/repo/$repo
